# データベースpbl2の作成
set names utf8;
drop database if exists pbl2;
create database pbl2;

# ユーザー pbl2に、パスワード pbl2 を設定し、データベース pbl2 に対するすべての権限を付与
grant all privileges on pbl2.* to pbl2@localhost identified by 'pbl2';

# データベース pbl2 を使用する
use pbl2;

# テーブル user の作成
create table user(
	ID	varchar(14) primary key,
	PASS	varchar(16) not null,
	Mail 	varchar(30) not null
	);
